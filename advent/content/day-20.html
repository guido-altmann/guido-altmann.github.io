public class GeschenkRepository : IGeschenkRepository {
    private readonly DbContext _context;

    public GeschenkRepository(DbContext context) {
        _context = context;
    }

    public async Task<IEnumerable<Geschenk>> GetAlleAsync() {
        return await _context.Geschenke.ToListAsync();
    }
}
